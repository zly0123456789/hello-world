char* StrCpy(char* src)
{
//1、计算src的长度 len
	int len=0;
	char* src1=src;
	while(*src1!="\0"){
		len++;
	}
//2、用malloc申请长度为 len+1的内存空间
	char* str=(char*) malloc(sizeof(char)*(len+1));
//3、将src字符串逐个字符复制到新申请的内存空间
	char *p1,*p2;
	p1=src;p2=str;
	for(;*p1!='\0';p1++,p2++){
		*p2=*p1;
	}
//4、新申请内存空间的最后一位放入'\0'
	*p2='\0';
//5、返回新申请的内存空间指针
	return *p2;
}
